public class Main {
    public static void main(String[] args) {
        String str="123,456,789";
        String[] arr=str.split(",");
        int max=0;
        for(int i=0;i<arr.length;i++){
            if(Integer.parseInt(arr[i])>max){
                max=Integer.parseInt(arr[i]);
            }
        }

    }
